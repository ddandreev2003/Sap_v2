🎛️ ГЛАВНЫЕ ПАРАМЕТРЫ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

--name              Какой пример генерировать
--num-steps         Количество шагов дифузии (20-100)
                    30: быстро (5-10 минут)
                    50: оптимально (10-15 минут)
                    80: качество (20-25 минут)

--num-seeds         Количество разных изображений (1-8)
--all               Генерировать все примеры
--list              Показать доступные примеры
--height            Высота изображения (default: 1024)
--width             Ширина изображения (default: 1024)
--device            cuda или cpu (default: cuda)
